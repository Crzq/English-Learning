$$
\sqrt{x}=>
\begin{cases}
{x}\geqslant0\\
\sqrt{x}\geqslant0
\end{cases}
$$


$$
\vert{x}\vert\geqslant0
$$


$$
\vert{x}\vert=
\begin{cases}
{x}& \text{x}\geqslant0\\
{-x}& \text{x}\leqslant0
\end{cases}
$$


$$
\sqrt{x^2}=
\begin{cases}
{x}& \text{x}\geqslant0\\
{-x}& \text{x}\leqslant0
\end{cases}
$$


$$
\sqrt{x}^2={x}
$$


$$
{x}^2=1=>{x}=\pm1
$$


$$
\sqrt{x^2}=1=>{x}=\pm1
$$


$$
\vert{x}\vert=1=>{x}=\pm1
